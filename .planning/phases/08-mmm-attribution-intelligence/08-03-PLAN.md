if st.button("Run MMM now"):
    import sqlite3
    from pathlib import Path as _Path
    result = fit_mmm(spend_arr, deposits_arr, settings.deposit_value_usd,
                     run_date=date.today().isoformat(), weeks_of_data=weeks)
    if result:
        # persist via sync sqlite3 directly (no DBClient/aiosqlite)
        with sqlite3.connect(str(settings.db_path)) as con:
            con.execute(
                "INSERT INTO mmm_results (...) VALUES (...)",
                result.to_dict(),
            )
        st.rerun()
    else:
        st.error("MMM fit failed — need more data or check logs.")
